// script.js (Logika untuk chatbot.html) let chatHistory = []; function handleEnter(event) { if (event.key === 'Enter') { kirimPesan(); } } function kirimPesan() { const input = document.getElementById('userInput'); const sendBtn = document.getElementById('sendBtn'); const pesan = input.value.trim(); if (!pesan) return; console.log("Mengirim pesan:", pesan); tampilkanPesan(pesan, 'user'); input.value = ''; sendBtn.disabled = true; sendBtn.textContent = '...'; chatHistory.push({ role: 'user', text: pesan }); // Tampilkan indikator berpikir const thinkingId = tampilkanPesanSedangBerpikir(); // Gunakan GET untuk menghindari CORS preflight const url = new URL(APPS_SCRIPT_URL); url.searchParams.set('action', 'chat'); url.searchParams.set('message', pesan); url.searchParams.set('history', JSON.stringify(chatHistory.slice(-6))); fetch(url.toString(), { method: 'GET' }) .then(response => response.text()) .then(text => { // Hapus indikator berpikir segera setelah respons diterima hapusPesanSedangBerpikir(thinkingId); try { const data = JSON.parse(text); if (data.success) { tampilkanPesan(data.reply, 'bot'); chatHistory.push({ role: 'bot', text: data.reply }); } else { tampilkanPesan('⚠️ Error: ' + (data.error || 'Gagal memproses'), 'bot'); } } catch (e) { console.error('Parse error:', e, 'Raw text:', text); tampilkanPesan('⚠️ Maaf, format respons tidak dikenali.', 'bot'); } }) .catch(error => { console.error('Fetch error:', error); hapusPesanSedangBerpikir(thinkingId); // Pastikan dihapus juga saat error tampilkanPesan('⚠️ Maaf, terjadi kesalahan koneksi.', 'bot'); }) .finally(() => { sendBtn.disabled = false; sendBtn.textContent = 'Kirim'; }); } // Fungsi untuk convert markdown ke HTML function markdownToHtml(text) { if (!text) return ''; text = text.replace(/\*\*(.*?)\*\*/g, '$1'); text = text.replace(/\*(.*?)\*/g, '$1'); text = text.replace(/`(.*?)`/g, '$1'); text = text.replace(/\n/g, '
'); return text; } function tampilkanPesan(teks, pengirim) { const chatBox = document.getElementById('chatBox'); const messageDiv = document.createElement('div'); messageDiv.className = 'message ' + pengirim; messageDiv.innerHTML = markdownToHtml(teks); chatBox.appendChild(messageDiv); chatBox.scrollTop = chatBox.scrollHeight; } // Fungsi untuk menampilkan pesan "Sedang berpikir..." function tampilkanPesanSedangBerpikir() { const chatBox = document.getElementById('chatBox'); const thinkingDiv = document.createElement('div'); thinkingDiv.className = 'message bot thinking-message'; thinkingDiv.id = 'thinking-' + Date.now(); thinkingDiv.innerHTML = `
Sedang berpikir
`; chatBox.appendChild(thinkingDiv); chatBox.scrollTop = chatBox.scrollHeight; return thinkingDiv.id; } // Fungsi untuk menghapus pesan "Sedang berpikir..." function hapusPesanSedangBerpikir(id) { const thinkingElement = document.getElementById(id); if (thinkingElement) { thinkingElement.remove(); } }
